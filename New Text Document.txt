function validatepricetable(e) {
    var keypressed = null;
    if (window.event){
        keypressed = window.event.keyCode;
    }
    else{ 
        keypressed = e.which;
    }
    if ((keypressed < 48 || keypressed > 57)&&keypressed!=46){
        if (keypressed == 8 || keypressed == 9){
            return;
        }
        return false;
    }
}
